# For-Loop

````powershell
for ($i = 0; $i -lt 100; $i++){
    write-host $i
}
````

# While-Loop

````powershell
while ($i -le 5) {
    Write-Host $i; $i++
}
````

# Do-While-Loop

````powershell
do {
    Write-Host $i; $i++
}
while ($i -le 5)
````

# Do-Until-Loop

````powershell
do {
    Write-Host $i; $i++
}
until ($i -gt 5)
````

# Foreach-Loop

````powershell
foreach($item in $collection) {
    Write-Host $item
}
````

# Foreach-Object-Loop

````powershell
Get-ADUser -Filter *| Foreach-Object {
    Write-Host $_
}
````

# Timer-Loop

````powershell
1..60 | foreach {write-host '.' -nonewline;sleep -Seconds 1}
````
